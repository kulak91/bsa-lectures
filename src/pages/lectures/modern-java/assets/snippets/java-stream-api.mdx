  public class Streams {

        //List of squared even numbers from 1 to 10 inclusive
        private static List<Integer> foreach() {
            List<Integer> result = new ArrayList<>();
            for (int i = 1; i <= 10; i++) {
                if (i % 2 != 0) continue;
                result.add(i * i);
            }

            return result;
        }
        //prints Starting stream, Collecting, Filtering! x 10
        private static List<Integer> stream() {
          System.out.println("Starting stream");
          var temp = Stream
                  //from 1 while i <= 10 with step i + 1
                  .iterate(1, i -> i <= 10, i -> i + 1)
                  //take only even numbers
                  .filter(i -> {
                      System.out.println("Filtering!");
                      return i % 2 == 0;
                  })
                  //square number
                  .map(i -> i * i);
          System.out.println("Collecting");
          //convert stream to list
          return temp.collect(Collectors.toList());

      }


  }
